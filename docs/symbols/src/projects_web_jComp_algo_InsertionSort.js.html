1 /**
  2 	Sort the input array using the insertion sort method.
  3 	@param {Array} obj An array of numbers to sort
  4 	@param {Boolean} asc Boolean value indicating the direction to sort. True for ascending. False for decending.
  5 */
  6 jComp.algo.insertionSort = function(obj, asc){
  7 	if(asc){
  8 		for(var i=1; i<obj.length; i++){
  9 			var current = obj[i];
 10 			var j=i-1;
 11 			while(j>=0 && obj[j]>current){
 12 				obj[j+1] = obj[j];
 13 				j--;
 14 			}
 15 			obj[j+1] = current;
 16 		}
 17 	}else{
 18 		for(var i=obj.length-2; i>=0; i--){
 19 			var current = obj[i];
 20 			var j=i+1;
 21 			while(j<=obj.length && obj[j]>current){
 22 				obj[j-1] = obj[j];
 23 				j++;
 24 			}
 25 			obj[j-1] = current;
 26 		}
 27 	}
 28 }
