1 /**
  2 	@class A generic set object
  3 */
  4 jComp.struct.Set = function(){
  5 	this._aryData = [];
  6 }
  7 
  8 /**
  9 	Iterate over the items in the set
 10 	@param {Function} f The function to execute for each item in the set. If the function returns false then the iterator will halt
 11 	@returns {Boolean} true if the iterator processed all items without being terminated. false if the iterator was told to terminate before completion
 12 */
 13 jComp.struct.Set.prototype.each = function(f){
 14 	var blnTerminated = false;
 15 	for(var i=0; i<this._aryData; i++){
 16 		if(!f(this, this._aryData[i])){
 17 			blnTerminated = true;
 18 			break;
 19 		}
 20 	}
 21 	return blnTerminated;
 22 }
 23 
 24 /**
 25 	Determine if the set is empty
 26 	@returns {Boolean} true if the set is empty, false if it is not
 27 */
 28 jComp.struct.Set.prototype.isEmpty = function(){
 29 	return this._aryData.length == 0;
 30 }
 31 
 32 /**
 33 	Get the number of items in the set
 34 	@returns {Integer} The number of items in the set
 35 */
 36 jComp.struct.Set.prototype.size = function(){
 37 	return this._aryData.length;
 38 }
 39 
 40 /**
 41 	Determine if the set contains this item
 42 	@param {Object} value The item to look for
 43 	@returns {Boolean} true if the set contains the item, false if it does not
 44 */
 45 jComp.struct.Set.prototype.contains = function(value){
 46 	for(var i=0; i<this._aryData.length; i++){
 47 		if(this._aryData[i] == value) return true;
 48 	}
 49 	return false;
 50 }
 51 
 52 /**
 53 	Insert an item into the set if the set does not already contain that item
 54 	@param {Object} value An item to be inserted into the set
 55 	@returns void
 56 */
 57 jComp.struct.Set.prototype.insert = function(value){
 58 	if(!this.contains(value)){
 59 		this._aryData.push(value);
 60 	}
 61 }
 62 
 63 /**
 64 	Union this set with another set. After the function, this set will have all of the elements of set that was passed in.
 65 	@param {jComp.struct.Set} otherSet The set with which to union
 66 	@return void
 67 */
 68 jComp.struct.Set.prototype.union = function(otherSet){
 69 	otherSet.each(function(caller, value){
 70 		caller.insert(value);
 71 	});
 72 }
