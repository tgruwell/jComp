1 /**
  2 	@class A queue for handling FIFO tasks
  3 */
  4 jComp.struct.Queue = function(){
  5 	this._aryData = [];
  6 }
  7 
  8 /**
  9 	Determine if the queue is empty
 10 	@returns {Boolean} true if the queue is empty, false if it is not
 11 */
 12 jComp.struct.Queue.prototype.isEmpty = function(){
 13 	return this._aryData.length == 0;
 14 }
 15 
 16 /**
 17 	Get the number of items in the queue
 18 	@returns {Integer} The number of items in the queue
 19 */
 20 jComp.struct.Queue.prototype.size = function(){
 21 	return this._aryData.length;
 22 }
 23 
 24 /**
 25 	Get the object at the front of the queue without removing it
 26 	@returns {Object} The object at the front of the queue. Does not remove the object
 27 */
 28 jComp.struct.Queue.prototype.front = function(){
 29 	return this._aryData[this._aryData.length-1];
 30 }
 31 
 32 /**
 33 	Remove the object at the front of the queue
 34 	@returns {Object} The object at the front of the queue that has just been removed
 35 */
 36 jComp.struct.Queue.prototype.pop = function(){
 37 	return this._aryData.pop();
 38 }
 39 
 40 /**
 41 	Add an object to the back of the queue
 42 	@param {Object} value The value to be added to the queue. Can be any type of object
 43 	@returns void
 44 */
 45 jComp.struct.Queue.prototype.push = function(value){
 46 	this._aryData.splice(0, 0, value);
 47 }
