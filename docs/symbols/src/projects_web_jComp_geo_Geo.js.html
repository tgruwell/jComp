1 /**
  2 	Determine the number of dimensions an object exists in.
  3 	@param {Object} obj An object to look at, for example a point or a line
  4 	@return {Integer} The number of dimensions an object exists in
  5 */
  6 jComp.geo.dimensions = function(obj){
  7 	return obj.length;
  8 }

 10 /** 
 11 	Determine the distance between two points. Assumes both points have the same number of dimensions
 12 	but bases it off of the number of dimensions point1 is in.
 13 	@param {jComp.geo.Point} point1 The first point
 14 	@param {jComp.geo.Point} point2 The second point
 15 	@return {Float} The distance between the two points
 16 */
 17 jComp.geo.distance = function(point1, point2){
 18 	if(point1.length == 3){
 19 		// 3 dimensional
 20 		
 21 		// We don't use Math.pow here because it is generally slower than just writing it out long hand.
 22 		// We also compute the differences only once. Most JS engines don't optimize enough to only do it once
 23 		// These two optimizations gives a 50% increase in performance in older FF (3.6) and a 30% increase in Chrome (v 15)
 24 		var d1 = point1[0]-point2[0];
 25 		var d2 = point1[1]-point2[1];
 26 		var d3 = point1[2]-point2[2];
 27 		return $f(Math.sqrt(d1*d1 + d2*d2 + d3*d3));
 28 		
 29 	}else if(point1.length == 2){
 30 		// 2 dimensional
 31 		
 32 		var d1 = point1[0]-point2[0];
 33 		var d2 = point1[1]-point2[1];
 34 		return $f(Math.sqrt(d1*d1 + d2*d2));
 35 		
 36 	}else if(point1.length > 3){
 37 		// N dimensional
 38 		var sum = 0;
 39 		for(var i=0; i<point1.length; i++){
 40 			var d = point1[i]-point2[i];
 41 			sum += d*d;
 42 		}
 43 		return $f(Math.sqrt(sum));
 44 		
 45 	}else if(point1.length == 1){
 46 		// 1 dimensional
 47 		return point1[0] = point2[0];
 48 	}
 49 	return 0;
 50 }
